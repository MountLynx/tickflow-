# C uses A[1] to pin A's first ever output. A is NOT a producer of C

# read A's history via A[1] without receiving a token from A.
#
# Run:
#   python -m tickflow run examples/pipeline.txt -b examples/pipeline_beh.py

B.body: transform
B-->C
